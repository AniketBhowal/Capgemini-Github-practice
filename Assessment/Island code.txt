class Solution {
    public int numIslands(char[][] g) {
        int count=0;
        for(int i=0;i<g.length;i++){
            for(int j=0;j<g[0].length;j++){
                if(g[i][j]=='1'){
                    infect(i,j,g);
                    System.out.println("Infection succesful at ["+i+","+j+"]");
                    count++;
                }
            }
        }
        System.out.println(Arrays.deepToString(g));
        return count;
    }
    
    public static void infect(int i, int j, char[][] g){
        
        g[i][j]='2';
        //up
        if(i!=0&&g[i-1][j]=='1'){
             infect(i-1,j,g);
        }
        //right
        if(j!=g[0].length-1&&g[i][j+1]=='1'){
             infect(i,j+1,g);
        }
        //down
        if(i!=g.length-1&&g[i+1][j]=='1'){
             infect(i+1,j,g);
        }
        //left
        if(j!=0&&g[i][j-1]=='1'){
             infect(i,j-1,g);
        }
    }
}
